\section{Introduction}

\begin{frame}{Introduction}
  \begin{columns}
    \column{0.6\textwidth}
      \textbf{Why Cambodia rice yield?}
      \begin{itemize}
        \item Rice is Cambodia's staple crop and largest agricultural export
        \item Yield fluctuations directly impact food security and rural livelihoods
        \item Country-aggregated annual data is publicly available from FAOSTAT and the World Bank
      \end{itemize}
      \vspace{0.5em}
      \textbf{Objectives}
      \begin{itemize}
        \item Build a merged dataset from FAOSTAT + World Bank climate data
        \item Train and compare 3 ML models
        \item Evaluate under both chronological and cross-validation protocols
      \end{itemize}
    \column{0.4\textwidth}
      \begin{block}{Tasks}
        \textbf{Regression:} Predict exact yield (kg/ha)\\[0.5em]
        \textbf{Classification:} Predict High or Low yield (above/below median)
      \end{block}
  \end{columns}
\end{frame}

\section{Dataset}

\begin{frame}{Dataset}
  \textbf{3 sources merged on year (1990--2024):}
  \vspace{0.1em}
  \begin{columns}
    \column{0.5\textwidth}
      \begin{itemize}
        \item \textbf{FAOSTAT} --- area harvested, production, yield
        \item \textbf{World Bank} --- annual mean temperature (\textdegree C)
        \item \textbf{World Bank} --- annual precipitation (mm)
      \end{itemize}
      \vspace{0.5em}
      \textit{Production was dropped: yield = production / area, so it would leak the target.}
    \column{0.5\textwidth}
      \begin{table}
        \centering
        \begin{tabular}{ll}
          \toprule
          \textbf{Feature} & \textbf{Unit} \\
          \midrule
          Area harvested   & ha \\
          Mean temperature & \textdegree C \\
          Precipitation    & mm \\
          \midrule
          \textbf{Target (reg.)} & kg/ha \\
          \textbf{Target (cls.)} & High / Low \\
          \bottomrule
        \end{tabular}
      \end{table}
  \end{columns}
  \vspace{0.1em}
  \begin{block}{Dataset size}
    \textbf{35 rows} $\cdot$ 3 features $\cdot$ 1 target \quad|\quad
    Chronological split: 28 train (1990--2017) / 7 test (2018--2024) \\
    Classification threshold: median computed on \textbf{training set only} (no leakage)
  \end{block}

\section{Proposed Methods}

\begin{frame}{Method 1 --- Linear Regression}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{itemize}
        \item Predicts yield as a weighted sum of features:
          \[ \hat{y} = \mathbf{w}^\top \mathbf{x} + b \]
        \item Fitted via Ordinary Least Squares (sklearn)
        \item Simple, interpretable baseline for regression
        \item No hyperparameters to tune
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{Why this model?}
        With only 35 samples, a simple model is less likely to overfit.
        Provides a strong baseline to compare against more complex approaches.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Method 2 --- Logistic Regression}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{itemize}
        \item Classifies yield as High (1) or Low (0):
          \[ \hat{p} = \sigma(\mathbf{w}^\top \mathbf{x} + b) \]
        \item Decision threshold: $\hat{p} > 0.5 \Rightarrow$ High
        \item Implemented with sklearn \texttt{LogisticRegression}
        \item Baseline for the classification task
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{Why this model?}
        Natural extension of linear regression to binary classification.
        Directly comparable to the neural network on the same task.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Method 3 --- Neural Network (PyTorch)}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{Architecture:} \quad $3 \rightarrow 8 \rightarrow 1$
      \begin{itemize}
        \item Input: 3 features
        \item Hidden: 8 neurons, ReLU activation
        \item Output: 1 neuron
          \begin{itemize}
            \item Regression: no activation, MSE loss
            \item Classification: Sigmoid, BCE loss
          \end{itemize}
        \item Optimiser: Adam (lr $= 0.1$)
        \item Epochs: 500, seed = 42
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{Why this model?}
        Can learn non-linear patterns. Kept small ($3{\to}8{\to}1$) to limit overfitting on 35 samples. Directly comparable to linear baselines.
      \end{block}
  \end{columns}
\end{frame}

\section{Results}

\begin{frame}{Results --- Regression (Chronological Split)}
  \begin{columns}
    \column{0.5\textwidth}
      \begin{table}
        \centering
        \caption{Test = 2018--2024}
        \begin{tabular}{lcc}
          \toprule
          \textbf{Model} & \textbf{RMSE} & \textbf{R$^2$} \\
          \midrule
          Predict train mean & 1188 & $-96.86$ \\
          Linear Regression  & \textbf{250.5} & $-3.35$ \\
          Neural Network     & 452.9 & $-13.22$ \\
          \bottomrule
        \end{tabular}
      \end{table}
      \vspace{0.5em}
      \begin{itemize}
        \item Linear Regression beats the naive baseline by $\sim$5$\times$
        \item But both models still have negative $R^2$
      \end{itemize}
    \column{0.5\textwidth}
      \begin{block}{Why negative $R^2$?}
        Test years (2018--2024) yield 3335--3644 kg/ha; train max is 3297. Test points sit \textbf{outside} the training range — models extrapolate. This is distribution shift, not a model-capacity failure.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Results --- Regression (5-fold Cross-Validation)}
  \begin{columns}
    \column{0.5\textwidth}
      \begin{table}
        \centering
        \caption{Mean $\pm$ std over 5 random folds}
        \begin{tabular}{lcc}
          \toprule
          \textbf{Model} & \textbf{RMSE} & \textbf{R$^2$} \\
          \midrule
          Linear Reg. & $\mathbf{164 \pm 30}$ & $\mathbf{0.92 \pm 0.05}$ \\
          Neural Net  & $214 \pm 92$         & $0.87 \pm 0.08$ \\
          \bottomrule
        \end{tabular}
      \end{table}
      \vspace{0.5em}
      \begin{itemize}
        \item When test points are interleaved with train: both models work very well
        \item Linear Regression edges out NN
      \end{itemize}
    \column{0.5\textwidth}
      \begin{block}{Take-away}
        Same data, two protocols, opposite stories. The chronological gap was about \emph{extrapolation}, not the models. Under fair CV, $R^2 \approx 0.92$.
      \end{block}
  \end{columns}
\end{frame}

\section{Conclusion}

\begin{frame}{Conclusion}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{Key findings:}
      \begin{itemize}
        \item Under cross-validation: Linear Regression $R^2 = 0.92$, beats NN
        \item Chronological negative $R^2$ = distribution shift, not capacity
        \item Classification on chronological split is degenerate; CV gives the real comparison (LR $\approx$ NN)
        \item Linear Reg. coefficients: \emph{area harvested} dominates (mechanisation effect)
      \end{itemize}
      \vspace{0.4em}
      \textbf{Future work:}
      \begin{itemize}
        \item Province-level data $\Rightarrow$ $\sim$25$\times$ more rows
        \item Add fertiliser, irrigation, flood data
        \item LSTM once data volume supports it
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{Take-away}
        On a 35-row dataset, the simpler model wins under fair evaluation, and how you split matters more than which model you pick.
      \end{block}
  \end{columns}
